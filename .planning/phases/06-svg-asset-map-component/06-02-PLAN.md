**Step 3 — Add `<script>` block with single shared timer:**
Add a `<script>` block (vanilla JS, NOT `<script is:inline>`) with:

```typescript
function initWorldMapClocks() {
  const clockEls = document.querySelectorAll('[data-clock]') as NodeListOf<SVGTextElement>;
  if (clockEls.length === 0) return;

  function formatClockTime(date: Date, iana: string): string {
    return date.toLocaleTimeString('en-US', {
      timeZone: iana,
      hour: 'numeric',
      minute: '2-digit',
      hour12: true,
    });
  }

  function updateAllClocks() {
    const now = new Date();
    clockEls.forEach(el => {
      const iana = el.dataset.clock!;
      el.textContent = formatClockTime(now, iana);
    });
  }

  // Initial update immediately
  updateAllClocks();

  // Single shared timer — per MAP-03, no per-zone intervals
  let timerId: number | null = setInterval(updateAllClocks, 1000);

  // Page Visibility API: pause when tab hidden, resume when visible
  document.addEventListener('visibilitychange', () => {
    if (document.hidden) {
      if (timerId !== null) {
        clearInterval(timerId);
        timerId = null;
      }
    } else {
      updateAllClocks(); // Immediate refresh on return
      if (timerId === null) {
        timerId = setInterval(updateAllClocks, 1000);
      }
    }
  });
}

// DOMContentLoaded dual-init pattern (matches TimezoneConverter)
if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', initWorldMapClocks);
} else {
  initWorldMapClocks();
}
```
